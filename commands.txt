#Boot to twrp tecovery

#masukan / push parted script into sdcard

adb push E:\arm64\parted /sdcard/

#start adb shell
adb shell
#move parted to sbin directory
cp /sdcard/parted /sbin/ && chmod 755 /sbin/parted
#umount data and sdcard
umount /data && umount /sdcard
#start parted
parted /dev/block/sda

#Hapus Partisi User Data(Menyesuaikan no Partisi) dengan Command rm

#Buat Partisi (256GB model)
mkpart esp fat32 6559MB 7000MB
mkpart win ntfs 7000MB 220GB
mkpart pe fat32 220GB 221GB
mkpart userdata ext4 221GB 253GB

#Buat Partisi (128GB model)
mkpart esp fat32 6559MB 7000MB
mkpart win ntfs 7000MB 100GB
mkpart pe fat32 100GB 101GB
mkpart userdata ext4 101GB 123GB

#Buat Partisi (64GB model)
mkpart esp fat32 6559MB 7000MB
mkpart win ntfs 7000MB 40GB
mkpart pe fat32 40GB 41GB
mkpart userdata ext4 41GB 59.1GB


#set esp on
set 1 esp on
#Quit
quit

#reboot to twrp

#format Partisi
mkfs.fat -F32 -s1 /dev/block/by-name/pe
mkfs.fat -F32 -s1 /dev/block/by-name/esp
mkfs.ntfs -f /dev/block/by-name/win
mke2fs -t ext4 /dev/block/by-name/userdata

#mount Partisi PE
mount /dev/block/by-name/pe /mnt

#Copy File PE ke Folder mnt di HP

#Masuk ke Mode Fastboot dan Boot ke UEFI
fastboot boot File EFI boot
